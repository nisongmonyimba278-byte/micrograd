\begin{frame}{The problem: fixed‑tree gradient generators}
\begin{columns}
\column{0.5\textwidth}
\begin{itemize}
\item Classic ``Christmas tree'' (Jeon \textit{et al.}, 2005)
\item 5 inlets, 4 mixing stages
\item Binomial concentration profile only
\item Manual design – no guarantee of optimality
\end{itemize}
\column{0.5\textwidth}
\includegraphics[width=\textwidth]{../docs/si/S6_christmas_tree_density.png}
\end{columns}
\end{frame}

\begin{frame}{Our approach: topology optimisation}
\begin{center}
\includegraphics[width=0.9\textwidth]{../docs/si/S12_gallery.pdf}
\end{center}
\begin{itemize}
\item Start from a blank rectangle
\item Evolve material distribution to match a \textbf{user‑specified} concentration profile
\item Coupled Brinkman flow + convection‑diffusion + adjoint sensitivity
\end{itemize}
\end{frame}

\begin{frame}{Key result – linear gradient generator}
\begin{columns}
\column{0.5\textwidth}
\includegraphics[width=\textwidth]{../figures/outlet_profile.pdf}
\column{0.5\textwidth}
\includegraphics[width=\textwidth]{../figures/bar_chart_resistance.pdf}
\end{columns}
\begin{itemize}
\item Topology‑optimised design \textbf{matches} the target profile
\item \textbf{~50\% lower} hydraulic resistance than the Christmas tree
\end{itemize}
\end{frame}

\begin{frame}{Generality – arbitrary target profiles}
\begin{center}
\includegraphics[width=0.85\textwidth]{../figures/gallery_target_profiles.pdf}
\end{center}
\begin{itemize}
\item Linear, sigmoid, double‑peak, stair‑step – all generated automatically
\item No pre‑prescribed architecture
\end{itemize}
\end{frame}

\begin{frame}{Validation}
\begin{columns}
\column{0.5\textwidth}
\begin{itemize}
\item Full Navier–Stokes simulation (body‑fitted mesh)
\item 3D extrusion validated
\item Uncertainty quantification (polynomial chaos)
\item Fabrication checks (min feature size $\ge$ 10 µm)
\end{itemize}
\column{0.5\textwidth}
\includegraphics[width=\textwidth]{../figures/3d_channel_render.png}
\end{columns}
\end{frame}

\begin{frame}{Why this matters for Ghana \& Africa}

\item No cleanroom needed for \textbf{design} – only for final fabrication
\item Open‑source Python pipeline – zero license cost
\item Summer 2025: workshops at UG \& Ashesi University
\end{itemize}
\end{frame}
